\documentclass[tikz, convert={density=300}]{standalone}
\usepackage{pgfplots}
\tikzstyle{load}=[ultra thick,-latex]
\tikzstyle{stress}=[-latex]
\tikzstyle{dim}=[latex-latex]
\tikzstyle{axis}=[-latex,black!55]
\begin{document}
    \begin{tikzpicture}[cm={-1,-1,1,0,(0,0)},x=3.85mm,z=-1cm]
		
        \coordinate (O) at (0,0,0);
        \draw[axis] (O) -- ++(3,0,0) node[right] {$x$};
        \draw[axis] (O) -- ++(0,3,0) node[above right] {$y$};
        \draw[axis] (O) -- ++(0,0,3) node[above] {$z$};	
        
        \def\xO{0}
        \def\yO{0}
        \def\zO{0}
        
        \def\vx{1.6}
        \def\vy{0.6}
        \def\vz{-0.2}
        
        \def\wx{-0.5}
        \def\wy{1.2}
        \def\wz{0.7}
        
        \fill[blue!30,opacity=0.6] (-\wx-\vx,-\wy-\vy,-\wz-\vz)-- (-\wx+\vx,-\wy+\vy,-\wz+\vz) -- (\wx+\vx,\wy+\vy,\wz+\vz) -- (\wx-\vx,\wy-\vy,\wz-\vz) -- (-\wx-\vx,-\wy-\vy,-\wz-\vz);
        
        \draw[load, red] (\xO,\yO,\zO) -- ++(\vx,\vy,\vz) node[right,xshift=0.3cm] {$\vec{v}$};	
        \draw[load, blue] (\xO,\yO,\zO) -- ++(\wx,\wy,\wz) node[right,xshift=0.3cm] {$\vec{w}$};	
        \draw[load, brown] (\xO,\yO,\zO) -- ++(\vy*\wz-\vz*\wy, \vz*\wx-\vx*\wz, \vx*\wy-\wx*\vy) node[right,xshift=0.3cm] {$\vec{v} \times \vec{w}$};	
    \end{tikzpicture}
\end{document}
